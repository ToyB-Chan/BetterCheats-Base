###
# Ensures that there is a global dummy country used by the bettercheats mod
#
# @return 					Saves global event target of dummy country as 'bc_base_global_dummy_country' if one has been created		
###
bc_base_ensure_dummy_country = {
	if = {
		limit = {
			not = {
				exists = event_target:bc_base_global_dummy_country
			}
		}

		create_country = {
			type = bc_base_global_dummy
			effect = {
				save_global_event_target_as = bc_base_global_dummy_country
			}
		}
	}
}
