###
# Grands scoped country host permissions
###
bc_grand_host_permission = {
	set_country_flag = BC_PERM_HOST
	log = "bc_base: granted [THIS.name] host permissions"
}

###
# Revokes scoped country host permissions
###
bc_revoke_host_permission = {
	remove_country_flag = BC_PERM_HOST
	log = "bc_base: revoked [THIS.name] host permissions"
}
