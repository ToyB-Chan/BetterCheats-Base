###
# Adds a bettercheats category to the country in scope
#
# @param	$ID$		 	Unique string used to identify this category 
# @param	$INDEX 			Number used to order categories inside the ui (lowest is first)
# @param	$LOC_TITLE$		Localization string of the category title
# @param	$LOC_DESC$		Localization string of the category description
# @return 					Saves event target of created category (leader type) as 'bc_base_create_category_return'
###
bc_base_create_category = {
	create_leader = {
		class = bc_base_dummy_leader
		immortal = yes
		name = CAT_$ID$
		effect = {
			set_leader_flag = BC_CAT
			set_leader_flag = BC_CAT_HEAD
			set_leader_flag = BC_CAT_ID_$ID$

			set_leader_flag = BC_CAT_ENABLED
			set_variable = {
				which = index
				value = $INDEX$
			}

			save_event_target_as = bc_base_create_category_return
		}
	}

	create_leader = {
		class = bc_base_dummy_leader
		immortal = yes
		name = $LOC_TITLE$
		effect = {
			set_leader_flag = BC_CAT
			set_leader_flag = BC_CAT_TITLE
			set_leader_flag = BC_CAT_ID_$ID$
		}
	}

	create_leader = {
		class = bc_base_dummy_leader
		immortal = yes
		name = $LOC_DESC$
		effect = {
			set_leader_flag = BC_CAT
			set_leader_flag = BC_CAT_DESC
			set_leader_flag = BC_CAT_ID_$ID$
		}
	}
}

###
# Returns a bettercheats category of the country in scope
#
# @param	$ID$ 			Unique string used to identify the category 
# @return 					Saves event target of category (leader type) as 'bc_base_get_category_return'
###
bc_base_get_category = {
	every_owned_leader = {
		limit = {
			bc_base_is_category_with_id = {ID = $ID$}
			bc_base_is_category_head = yes
		}

		save_event_target_as = bc_base_get_category_return
	}
}

###
# Clears all bettercheats categories of the country in scope
###
bc_base_clear_all_categories = {
	every_owned_leader = {
		limit = {
			bc_base_is_category = yes
		}

		kill_leader = {
			show_notification = no
		}
	}
}

###
# Enables the category in scope
###
bc_base_enable_category = {
	set_leader_flag = BC_CAT_ENABLED
}

###
# Disables the category in scope
###
bc_base_disable_category = {
	remove_leader_flag = BC_CAT_ENABLED
}
